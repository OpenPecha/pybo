ལ་ རང་རང་ གི་ མིང་ གི་ བཏགས་ དོན་ དང་ རྒྱུ་མཚན་ སོགས་ ཡོད་པ་ མ་ཟད ། མིང་ཚིག་ རང་རང་ ལ་ རང་རང་ གི་ གོ་དོན་ དང་ ངེས་ཚིག ། མཚན་ཉིད་ རེ འང་ ཡོད་པ་ རེད ། མིང་ གི་ བཏགས་ དོན་ ཡོད་པ འི་ མིང་ ཚང་མ་ རྗེས་གྲུབ་ ཀྱི་ མིང་ ཡིན་ ཕྱིར ། མིང་ཚིག་ གི་ གོ་དོན་ མ་བཀླག་ ཀྱང་ དེ འི་ མིང་ གི་ བཏགས་ དོན་ དང་ དེ འི་ ནུས་པ ། རང་བཞིན་ ཆོས་ཉིད་ གང་རུང་ རང་ཤུགས་ ཀྱིས་ རྟོགས་ ཡོང་བ་ རེད ། རྗེས་གྲུབ་ ཀྱི་ མིང་ རྣམས་ ནི་ འདོད་རྒྱལ་ གྱི་ མིང་ མང་ དུ་ འཕེལ་བ འི་ རྨང་གཞི འི་ འོག་ ནས་ བྱུང་བ་ ཞིག་ ཡིན་ ལ་ རྒྱན་ཚིག་ དང་ བདག་སྒྲ་ མངོན་བརྗོད་ ལ་སོགས་པ འང་ རྗེས་གྲུབ་ ཀྱི་ མིང་ གི་ ཁོངས་ སུ་ གཏོགས་པ་ རེད ། འདོད་རྒྱལ་ གྱི་ མིང་ ཕུད་ གཞན་ མིང་བརྡ་ ཡོད་ཚད་ ཀྱང་ རྗེས་གྲུབ་ ཀྱི་ མིང་ ཡིན་ ལ ། རྗེས་གྲུབ་ ཀྱི་ མིང་ ཡོད་ཚད་ ལ་ རང་རང་ གི་ མིང་ གི་ བཏགས་ དོན་ ཡོད་པ ས་ དོན་གཅིག་ ཀྱང་ མཚན་ཉིད་ མི་འདྲ་བ་ རེ་ འཇོག་ སྲོལ་ ཡོད་པ་ རེད ། བཞི་བ ། དོན་གཅིག་པ འི་ མིང་ བཤད་པ ། དོན་གཅིག་ ཞེས་པ་ ནི་ མིང་ཚིག་ མི་འདྲ་བ་ མང་པོ འི་ དོན་གཅིག་ སྟོན་ གྱི་ ཡོད་པ་ ཞིག་ ལ་ ཟེར་ ཏེ་ དཔེར་ན་ རྒྱལ་བ་ དང་ བདེ་གཤེགས ། ཀུན་མཁྱེན ། དཔལ་སྟུག ། རྐང་གཉིས་གཙོ ། བཅོམ་ལྡན་འདས ། སྟོབས་བཅུ་ མངའ་བ་ ལ་སོགས་པ འི་ མིང་ཚིག་ མང་པོ་ ཞིག་ གི་ དོན་ སངས་རྒྱས་ སྟོན་ ཀྱི་ ཡོད་པ་ ལྟ་བུ འོ ། དེ་ལྟ་བུ འི་ དོན་གཅིག་ ལ་ མིང་ མང་པོ་ ཡོད་པ་ དེ་དག་ རེ་རེ་ ལ འང་ རང་རང་ གི་ མིང་ གི་ བཏགས་ དོན་ དང་ གོ་དོན་ ཡོད་ དེ་ དཔེར་ན་ ཆུ་འཛིན་ ལ་ ཆུ་འཛིན་ གྱི་ མིང་བཏགས་ དོན་ དང་ མཚན་ཉིད་ ཡོད་པ་ དང་ ཚ་ཟེར་འཇོམས་ ལ་ ཚ་ཟེར་འཛོམས་ ཀྱི་ བཏགས་ དོན་ མཚན་ཉིད་ ཡོད་པ་ དང་ ཆུ་ཁུར་ ལ་ ཆུ་ཁུར་ གྱི་ མིང་
